                Question 1
                
                Write a program in PL/SQL to print the prime numbers between 1 to 100.
                
                Your Answer


                DECLARE
                
                  i NUMBER(3);
                
                  j NUMBER(3);
                
                BEGIN
                
                dbms_output.Put_line('The prime numbers are:');
                
                dbms_output.new_line;
                
                  i := 2;
                
                  LOOP
                
                    j := 2;
                
                    LOOP
                
                      EXIT WHEN( ( MOD(i, j) = 0 )
                
                            OR ( j = i ) );
                
                      j := j + 1;
                
                    END LOOP;
                
                    IF( j = i )THEN
                
                     dbms_output.Put(i||'  ');  
                
                    END IF;
                
                    i := i + 1;
                
                    exit WHEN i = 100;
                
                  END LOOP;
                
                dbms_output.new_line;
                
                END;
                
                /
                
                Question 2

                Write a PL/SQL program to display the description against a grade according to the marks achieved by students in their final exams.
                
                Your Answer


                DECLARE
                
                  grd CHAR(1);
                
                 BEGIN
                
                  -- Accept value for grade
                
                  grd := '&new_grd';
                
                 CASE grd
                
                  WHEN 'A' THEN dbms_output.Put_line('Your Grade is: Outstanding');
                
                  WHEN 'B' THEN dbms_output.Put_line('Your Grade is: Excellent');
                
                  WHEN 'C' THEN dbms_output.Put_line('Your Grade is: Very Good');
                
                  WHEN 'D' THEN dbms_output. Put_line('Your Grade is: Average');
                
                  WHEN 'F' THEN dbms_output.Put_line('Your Grade is: Poor');
                
                  ELSE dbms_output.Put_line('No such grade in the list.');
                
                 END CASE;
                
                 END;
